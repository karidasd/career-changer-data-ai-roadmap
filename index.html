import streamlit as st import pandas as pd st.title("📊 Sales Dashboard") df = pd.read_csv("sales_data.csv") # Filter Sidebar region = st.sidebar.selectbox( "Select Region", ["All", "North", "South", "East"] ) if region != "All": df = df[df["Region"] == region] # Metrics row st.metric("Total Revenue", df["Revenue"].sum()) st.metric("Total Orders", len(df)) # Chart category_sums = df.groupby("Category")["Revenue"].sum() st.bar_chart(category_sums)
